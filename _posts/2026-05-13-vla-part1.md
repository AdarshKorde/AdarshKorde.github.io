<!-- The Canvas where the robot is drawn -->
<canvas id="robotCanvas" width="500" height="300" style="border: 1px solid #ddd; border-radius: 8px; background: white;"></canvas>

<!-- The Logic -->
<script>
  const canvas = document.getElementById('robotCanvas');
  const ctx = canvas.getContext('2d');
  const j1Input = document.getElementById('j1');
  const j2Input = document.getElementById('j2');

  function drawRobot() {
    // Clear previous drawing
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // Convert degrees to radians
    let theta1 = j1Input.value * (Math.PI / 180);
    let theta2 = j2Input.value * (Math.PI / 180);

    // Arm properties
    let baseX = 250, baseY = 250;
    let L1 = 100, L2 = 80;

    // Kinematics math
    let joint1X = baseX + L1 * Math.cos(theta1 - Math.PI/2);
    let joint1Y = baseY + L1 * Math.sin(theta1 - Math.PI/2);
    
    let endX = joint1X + L2 * Math.cos(theta1 + theta2 - Math.PI/2);
    let endY = joint1Y + L2 * Math.sin(theta1 + theta2 - Math.PI/2);

    // Draw Link 1
    ctx.beginPath(); ctx.moveTo(baseX, baseY); ctx.lineTo(joint1X, joint1Y);
    ctx.lineWidth = 12; ctx.strokeStyle = '#2c3e50'; ctx.stroke();

    // Draw Link 2
    ctx.beginPath(); ctx.moveTo(joint1X, joint1Y); ctx.lineTo(endX, endY);
    ctx.lineWidth = 8; ctx.strokeStyle = '#e74c3c'; ctx.stroke();

    // Draw Joints (Circles)
    ctx.beginPath(); ctx.arc(baseX, baseY, 10, 0, Math.PI*2); ctx.fillStyle = '#34495e'; ctx.fill();
    ctx.beginPath(); ctx.arc(joint1X, joint1Y, 8, 0, Math.PI*2); ctx.fillStyle = 'white'; ctx.fill(); ctx.stroke();
    ctx.beginPath(); ctx.arc(endX, endY, 5, 0, Math.PI*2); ctx.fillStyle = '#e74c3c'; ctx.fill();
  }

  // Listen for slider movement
  j1Input.addEventListener('input', drawRobot);
  j2Input.addEventListener('input', drawRobot);
  
  // Draw the initial state
  drawRobot();
</script>
